# minikube

# Installation

curl -LO https://storage.googleapis.com/minikube/releases/latest/minikube-latest.x86_64.rpm
sudo rpm -Uvh minikube-latest.x86_64.rpm

# Start your cluster

useradd k8s
usermod -a -G docker k8s
passwd
su - k8s

# 启动集群
minikube start
# 启动集群
minikube start \
--driver=docker \
--kubernetes-version=v1.23.8 \
--image-mirror-country='cn' \
--image-repository='registry.cn-hangzhou.aliyuncs.com/google_containers'

#  停止集群
minikube stop

# 获取集群状态
minikube status

# 删除集群
minikube delete

# 查看集群IP

# 查看集群日志
minikube logs

# Deploy applications

Create a sample deployment and expose it on port 80:

kubectl create deployment hello-minikube --image=docker.io/nginx:1.23
kubectl expose deployment hello-minikube --type=NodePort --port=80
1
2

# 核心概念及操作

https://kubernetes.io/docs/home/ (opens new window)

资源创建方式有两种：

  • 命令行
  • YAML

# kubectl

kubectl 命令行的语法格式如下：

kubectl --help
kubectl [command] [TYPE] [NAME] [flags]
1
2

  • command：子命令，用于操作 k8s 集群资源对象的命令，例如 create、get、describe、delete。

    更多命令见：https://kubernetes.io/docs/reference/generated/kubectl/kubectl-commands (opens new window)

  • TYPE：k8s 集群中资源对象的类型，区分大小写，能以单数、复数或者简写形式表示

    • pod（po）

    • service（svc）

    • replication controller（rc）

    • deployment（deploy）

    • replica set（rs）

  • NAME：k8s 集群中资源对象的名称，区分大小写。如果不指定名称，系统则返回属于 TYPE 的全部资源对象列表

  • flags：kubectl 子命令的可选参数

    kubectl 每个子命令（如 create、get、describe、delete 等）还有特定的 flags 参数，可以通过

    kubectl [command] --help
kubectl [command] [TYPE] --help
    1
    2

    命令查看。

# Namespace

Namespace 默认只隔离资源，不隔离网络。语法格式：

kubectl [command] ns [NAME] [flags]
kubectl [command] ns --help
1
2

示例：

kubectl get ns
kubectl create ns dev
kubectl delete ns dev
# 查看已存在命名空间的yaml配置
kubectl get ns dev -o=yaml
# 导出已存在命名空间的yaml配置
kubectl get ns dev -o=yaml > dev-namespace.yaml
# 导出不存在命名空间的yaml配置
kubectl create ns test -o=yaml --dry-run=client > test-namespace.yaml

# Pod

Pod 运行中的一组容器，Pod是kubernetes的最小单位。

语法格式：

kubectl [command] pod [NAME] [flags]
kubectl [command] pod --help
1
2

示例：

# 查看所有命名空间的pod
kubectl get pod -A
# 查看默认名称空间下的pod
kubectl get pod
# 查看默认名称空间下的pod（显示更加详细的信息）
kubectl get pod -owide
# 监听pod的变化
kubectl get pod -w

# Create and run a particular image in a pod.
kubectl run my-nginx --image=docker.io/nginx:1.23 --port=80

# Show details of a specific resource or group of resources.
kubectl describe pod my-nginx

# Print the logs for a container in a pod or specified resource. If the pod has only one container, the container name is optional.
kubectl logs my-nginx

# Execute a command in a container.
# kubectl exec (POD | TYPE/NAME) [-c CONTAINER] [flags] -- COMMAND [args...]
kubectl exec -it my-nginx -- /bin/bash

kubectl delete pod my-nginx

# Deployment

Deployment 控制 Pod，使Pod拥有多个副本，自愈、扩容等功能。

语法格式：

kubectl [command] deploy [NAME] [flags]
kubectl [command] deploy --help
1
2

示例：

kubectl get deployment

kubectl create deployment my-nginx --image=docker.io/nginx:1.23
kubectl create deployment my-nginx --image=docker.io/nginx:1.23 --replicas=3

kubectl expose deployment my-nginx --type=NodePort --port=80

kubectl edit deployment my-nginx

# 扩容
kubectl scale --current-replicas=3 --replicas=4 deployment/my-nginx
# 缩容
kubectl scale --current-replicas=4 --replicas=2 deployment/my-nginx
# 滚动更新/不停机维护
kubectl set image deployment/my-nginx nginx=nginx:1.9.1 --record
# 查看历史记录
kubectl rollout history deployment/my-nginx
# 版本回退
kubectl rollout undo deployment/my-nginx --to-revision=1

kubectl delete deployment my-nginx
kubectl delete -n default deployment my-nginx

kubectl get deployment my-nginx -o=yaml
kubectl get deployment my-nginx -o=yaml > deployment.yaml
kubectl create deployment my-nginx --image=docker.io/nginx:1.23 --replicas=3 -o=yaml --dry-run=client > deployment.yaml

# Service

Service ，将一组Pods公开为网络服务的抽象方法，Pod的服务发现和负载均衡。

# Expose a resource as a new Kubernetes service.
# type: Type for this service: ClusterIP, NodePort, LoadBalancer, or ExternalName. Default is 'ClusterIP'.
# --type=ClusterIP: 仅仅集群内访问
# --type=NodePort: 集群外可以访问
# --port: The port that the service should serve on.
# --target-port: Name or number for the port on the container that the service should direct traffic to
kubectl expose pod my-pod-1 --port=10080 --target-port=80 --type=NodePort
kubectl expose pod my-nginx --port=10080 --target-port=80 --type=NodePort -o=yaml --dry-run=client > service.yaml

# Q&A

# ClusterIP和NodePort

ClusterIP，仅允许集群内访问

NodePort，在集群外可以访问，The range of valid ports is 30000-32767

# targetPort、port、nodePort

同一个pod中，不同container之间可以互相访问

apiVersion: v1
kind: Pod
metadata:
  name: my-pod
  namespace: default
spec:
  containers:
  - image: docker.io/nginx:1.23
    name: my-nginx
    ports:
    - containerPort: 80  # 与Service的targetPort值一致
  - image: tomcat
    name: my-tomcat
    ports:
    - containerPort: 8080 # 与Service的targetPort值一致

    protocol: TCP
    targetPort: 80  # container暴露出来的port，同一个pod中，不同container之间可以互相访问
    nodePort: 30080 # 与type: NodePort一起使用
  type: NodePort    # type设置为NodePort，就可以让外网访问了
  selector:
    run: my-nginx   # 通过selector来寻找相同label的pod
